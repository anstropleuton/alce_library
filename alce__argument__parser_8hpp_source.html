286[[nodiscard]] inline constexpr auto get_argument_type(std::string_view argument)
287{
288 if (argument.size() == 0) return argument_type::empty;
289
290 if (argument.starts_with("--"))
291 {
292 if (argument.size() > 2) return argument_type::long_option;
293 else return argument_type::double_hyphen;
294 }
295
296 if (argument.starts_with("-"))
297 {
298 if (argument.size() > 1) return argument_type::short_option;
299 else return argument_type::single_hyphen;
300 }
301
302 if (argument.starts_with("/"))
303 {
304 return argument_type::microsoft_switch;
305 }
306
307 return argument_type::regular_argument;
308}

675struct styled_padding {
676
680 styled_text first;
681
685 styled_text mid;
686
690 styled_text last;
691
699 std::size_t width;
700
707 [[nodiscard]] inline constexpr auto str(std::size_t subtract = 0) const
708 {
709 using namespace std::string_literals;
710 if (subtract > width)
711 {
712 return ""s;
713 }
714
715 std::size_t actual_size = width - subtract;
716 if (actual_size == 0)
717 {
718 return ""s;
719 }
720 if (actual_size == 1)
721 {
722 return mid.style(mid.value);
723 }
724 if (actual_size == 2)
725 {
726 return first.style(first.value)
727 + last.style(last.value);
728 }
729 return first.style(first.value)
730 + mid.style(sm::repeat(mid.value, actual_size))
731 + last.style(last.value);
732 }
733
740 [[nodiscard]] inline constexpr auto size(std::size_t subtract = 0) const
741 {
742 if (subtract > width)
743 {
744 return 0zu;
745 }
746
747 std::size_t actual_size = width - subtract;
748 if (actual_size == 0)
749 {
750 return 0zu;
751 }
752 if (actual_size == 1)
753 {
754 return mid.value.size();
755 }
756 if (actual_size == 2)
757 {
758 return first.value.size()
759 + last.value.size();
760 }
761 return first.value.size()
762 + mid.value.size() * actual_size
763 + last.value.size();
764 }
765
772 [[nodiscard]] inline constexpr auto m_str(std::size_t subtract = 0) const
773 {
774 return measured_string(str(subtract), size(subtract));
775 }
