293[[nodiscard]] inline constexpr auto get_argument_type(std::string_view argument)
294{
295 if (argument.size() == 0) return argument_type::empty;
296
297 if (argument.starts_with("--"))
298 {
299 if (argument.size() > 2) return argument_type::long_option;
300 else return argument_type::double_hyphen;
301 }
302
303 if (argument.starts_with("-"))
304 {
305 if (argument.size() > 1) return argument_type::short_option;
306 else return argument_type::single_hyphen;
307 }
308
309 if (argument.starts_with("/"))
310 {
311 return argument_type::microsoft_switch;
312 }
313
314 return argument_type::regular_argument;
315}

688struct styled_padding {
689
693 styled_text first;
694
698 styled_text mid;
699
703 styled_text last;
704
712 std::size_t width;
713
720 [[nodiscard]] inline constexpr auto str(std::size_t subtract = 0) const
721 {
722 using namespace std::string_literals;
723 if (subtract > width)
724 {
725 return ""s;
726 }
727
728 std::size_t actual_size = width - subtract;
729 if (actual_size == 0)
730 {
731 return ""s;
732 }
733 if (actual_size == 1)
734 {
735 return mid.style(mid.value);
736 }
737 if (actual_size == 2)
738 {
739 return first.style(first.value)
740 + last.style(last.value);
741 }
742 return first.style(first.value)
743 + mid.style(sm::repeat(mid.value, actual_size))
744 + last.style(last.value);
745 }
746
753 [[nodiscard]] inline constexpr auto size(std::size_t subtract = 0) const
754 {
755 if (subtract > width)
756 {
757 return 0zu;
758 }
759
760 std::size_t actual_size = width - subtract;
761 if (actual_size == 0)
762 {
763 return 0zu;
764 }
765 if (actual_size == 1)
766 {
767 return mid.value.size();
768 }
769 if (actual_size == 2)
770 {
771 return first.value.size()
772 + last.value.size();
773 }
774 return first.value.size()
775 + mid.value.size() * actual_size
776 + last.value.size();
777 }
778
785 [[nodiscard]] inline constexpr auto m_str(std::size_t subtract = 0) const
786 {
787 return measured_string(str(subtract), size(subtract));
788 }
