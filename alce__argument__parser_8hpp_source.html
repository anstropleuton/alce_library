285[[nodiscard]] inline constexpr auto get_argument_type(std::string_view argument)
286{
287 if (argument.size() == 0) return argument_type::empty;
288
289 if (argument.starts_with("--"))
290 {
291 if (argument.size() > 2) return argument_type::long_option;
292 else return argument_type::double_hyphen;
293 }
294
295 if (argument.starts_with("-"))
296 {
297 if (argument.size() > 1) return argument_type::short_option;
298 else return argument_type::single_hyphen;
299 }
300
301 if (argument.starts_with("/"))
302 {
303 return argument_type::microsoft_switch;
304 }
305
306 return argument_type::regular_argument;
307}

674struct styled_padding {
675
679 styled_text first;
680
684 styled_text mid;
685
689 styled_text last;
690
698 std::size_t width;
699
706 [[nodiscard]] inline constexpr auto str(std::size_t subtract = 0) const
707 {
708 using namespace std::string_literals;
709 if (subtract > width)
710 {
711 return ""s;
712 }
713
714 std::size_t actual_size = width - subtract;
715 if (actual_size == 0)
716 {
717 return ""s;
718 }
719 if (actual_size == 1)
720 {
721 return mid.style(mid.value);
722 }
723 if (actual_size == 2)
724 {
725 return first.style(first.value)
726 + last.style(last.value);
727 }
728 return first.style(first.value)
729 + mid.style(sm::repeat(mid.value, actual_size))
730 + last.style(last.value);
731 }
732
739 [[nodiscard]] inline constexpr auto size(std::size_t subtract = 0) const
740 {
741 if (subtract > width)
742 {
743 return 0zu;
744 }
745
746 std::size_t actual_size = width - subtract;
747 if (actual_size == 0)
748 {
749 return 0zu;
750 }
751 if (actual_size == 1)
752 {
753 return mid.value.size();
754 }
755 if (actual_size == 2)
756 {
757 return first.value.size()
758 + last.value.size();
759 }
760 return first.value.size()
761 + mid.value.size() * actual_size
762 + last.value.size();
763 }
764
771 [[nodiscard]] inline constexpr auto m_str(std::size_t subtract = 0) const
772 {
773 return measured_string(str(subtract), size(subtract));
774 }
