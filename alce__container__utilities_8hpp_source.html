1818[[nodiscard]] inline constexpr auto split_occ_seq(
1819 const Container &container,
1820 const NestedContainer &patterns
1821)
1822{
1823 // Again couldn't find standard library to do this heavy job, so...
1824 result_container_nested<Container> result;
1825 auto it = container.begin();
1826
1827 while (it != container.end())
1828 {
1829 auto next_it = container.end();
1830 std::size_t pattern_size = (std::size_t)-1;
1831 for (auto &pattern : patterns)
1832 {
1833 // Use std::search instead of std::find_first_of to find sequence
1834 auto tmp = std::search(it, container.end(), pattern.begin(),
1835 pattern.end());
1836
1837 if (next_it > tmp)
1838 {
1839 next_it = tmp;
1840 pattern_size = pattern.size();
1841 }
1842 }
1843
1844 result.emplace_back(result_container<Container>(it, next_it));
1845 it = next_it;
1846 if (it != container.end()) it += pattern_size;
1847 }
1848
1849 return result;
