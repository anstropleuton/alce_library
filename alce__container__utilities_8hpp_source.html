1724template<cu_compatible Container, cu_compatible_nested NestedContainer>
1725[[nodiscard]] inline constexpr auto split_occ_seq(
1726 const Container &container,
1727 const NestedContainer &patterns
1728)
1729{
1730 // Again couldn't find standard library to do this heavy job, so...
1731 result_container_nested<Container> result;
1732 auto it = container.begin();
1733
1734 while (it != container.end())
1735 {
1736 auto next_it = container.end();
1737 std::size_t pattern_size = (std::size_t)-1;
1738 for (auto &pattern : patterns)
1739 {
1740 // Use std::search instead of std::find_first_of to find sequence
1741 auto tmp = std::search(it, container.end(), pattern.begin(),
1742 pattern.end());
1743
1744 if (next_it > tmp)
1745 {
1746 next_it = tmp;
1747 pattern_size = pattern.size();
1748 }
1749 }
1750
1751 result.emplace_back(result_container<Container>(it, next_it));
1752 it = next_it;
1753 if (it != container.end()) it += pattern_size;
1754 }
1755
1756 return result;
1757}
