1816[[nodiscard]] inline constexpr auto split_occ_seq(
1817 const container &ctr,
1818 const nested_container &patterns
1819)
1820{
1821 // Again couldn't find standard library to do this heavy job, so...
1822 result_container_nested<container> result;
1823 auto it = ctr.begin();
1824
1825 while (it != ctr.end())
1826 {
1827 auto next_it = ctr.end();
1828 std::size_t pattern_size = (std::size_t)-1;
1829 for (auto &pattern : patterns)
1830 {
1831 // Use std::search instead of std::find_first_of to find sequence
1832 auto tmp = std::search(it, ctr.end(), pattern.begin(),
1833 pattern.end());
1834
1835 if (next_it > tmp)
1836 {
1837 next_it = tmp;
1838 pattern_size = pattern.size();
1839 }
1840 }
1841
1842 result.emplace_back(result_container<container>(it, next_it));
1843 it = next_it;
1844 if (it != ctr.end()) it += pattern_size;
1845 }
1846
1847 return result;
