97[[nodiscard]] inline constexpr auto to_string(
98 const Container &container,
99 Converter converter,
100 std::string_view separator = ", ",
101 std::string_view prefix = "",
102 std::string_view suffix = ""
103)
104{
105 auto transformer = [&](const cu::value_type<Container> &element)
106 {
108 return std::string(prefix) + converter(element) + std::string(suffix);
109 };
110
111 return std::views::transform(container, transformer)
112 | std::views::join_with(std::string(separator))
113 | std::ranges::to<std::string>();
114}

230[[nodiscard]] inline constexpr auto word_wrap(
231 std::string_view string,
232 std::size_t width,
233 bool force = false,
234 std::string_view delims = " \t\r\n\f\v\b"
235)
236{
237 result_string_nested lines = {};
238
239 // Functions expect inclusive width, and also works as a measure to have at
240 // least one as width (as a side effect)
241 width++;
242
243 while (string.size() > width)
244 {
245 // First non-delim character before or at width
246 auto pos = string.substr(0, width).find_last_of(delims);
247 if (pos == std::string_view::npos)
248 {
249 // Split without consuming character
250 if (force)
251 {
252 pos = string.substr(0, width).size() - 1;
253 auto line = string.substr(0, pos);
254 string = string.substr(pos);
255 lines.emplace_back(std::string(line));
256 continue;
257 }
258
259 // If not, first non-delim character after width
260 pos = string.substr(width + 1).find_first_of(delims);
261
262 // If still not, eof reached? Prevent overflow
263 if (pos == std::string_view::npos) pos--;
264 else pos += width + 1;
265 }
266
267 auto line = string.substr(0, pos);
268 string = string.substr(pos + 1);
269 lines.emplace_back(std::string(line));
270 }
271
272 if (!string.empty())
273 {
274 lines.emplace_back(std::string(string));
275 }
276
277 return lines;
278}
