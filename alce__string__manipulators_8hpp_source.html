95template<cu::cu_compatible container, typename converter>
96requires(!std::is_same_v<converter, std::string>)
97[[nodiscard]] inline constexpr auto to_string(
98 const container &ctr,
99 converter conv,
100 std::string_view separator = ", ",
101 std::string_view prefix = "",
102 std::string_view suffix = ""
103)
104{
105 auto transformer = [&](const cu::value_type<container> &element)
106 {
108 return std::string(prefix) + conv(element) + std::string(suffix);
109 };
110
111 return std::views::transform(ctr, transformer)
112 | std::views::join_with(std::string(separator))
113 | std::ranges::to<std::string>();
114}

240[[nodiscard]] inline constexpr auto word_wrap(
241 std::string_view string,
242 std::size_t width,
243 bool force = false,
244 std::string_view delims = " \t\r\n\f\v\b"
245)
246{
247 result_string_nested lines = {};
248
249 // Functions expect inclusive width, and also works as a measure to have at
250 // least one as width (as a side effect)
251 width++;
252
253 while (string.size() > width)
254 {
255 // First non-delim character before or at width
256 auto pos = string.substr(0, width).find_last_of(delims);
257 if (pos == std::string_view::npos)
258 {
259 // Split without consuming character
260 if (force)
261 {
262 pos = string.substr(0, width).size() - 1;
263 auto line = string.substr(0, pos);
264 string = string.substr(pos);
265 lines.emplace_back(std::string(line));
266 continue;
267 }
268
269 // If not, first non-delim character after width
270 pos = string.substr(width + 1).find_first_of(delims);
271
272 // If still not, eof reached? Prevent overflow
273 if (pos == std::string_view::npos) pos--;
274 else pos += width + 1;
275 }
276
277 auto line = string.substr(0, pos);
278 string = string.substr(pos + 1);
279 lines.emplace_back(std::string(line));
280 }
281
282 if (!string.empty())
283 {
284 lines.emplace_back(std::string(string));
285 }
286
287 return lines;
288}
